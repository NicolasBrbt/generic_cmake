set (Test_SOURCES "")
set (Test_HEADERS"")

file(GLOB Test_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp" "${CMAKE_CURRENT_SOURCE_DIR}/src/*.c")
file(GLOB Test_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}/include/*.hpp" "${CMAKE_CURRENT_SOURCE_DIR}/include/*.h")

add_library(Test STATIC ${Test_SOURCES} ${Test_HEADERS})

target_include_directories(Test PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include)

add_dependencies(Test copy_headers)
